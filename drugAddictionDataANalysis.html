def lowerupper(col):
    q1 = np.quantile(col, .25)
    q3 = np.quantile(col, .75)
    inter = q3 - q1
    lowerbound = q1 - (inter * 1.5)
    upperbound = q3 + (inter * 1.5)
    outlier = []
    
    for x in col:
        if (x > upperbound) or (x < lowerbound):
            outlier.append(x)
    
    return outlier
